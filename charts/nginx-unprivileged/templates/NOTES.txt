1. Forward traffic to the pod locally:
  export POD=$(kubectl get pods -l app.kubernetes.io/name={{ include "app.name" . }},app.kubernetes.io/instance={{ .Release.Name }} -o jsonpath="{.items[0].metadata.name}")
  kubectl port-forward $POD 8080:{{ .Values.service.targetPort }}

Then open http://127.0.0.1:8080/
